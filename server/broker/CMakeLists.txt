## INC & SOURCE DIRS
set(SFINDER_BROKERSERVER_INC_DIR ${CMAKE_CURRENT_SOURCE_DIR})
set(SFINDER_BROKERSERVER_SRC_DIR ${CMAKE_CURRENT_SOURCE_DIR})
message (STATUS "SFINDER_BROKERSERVER_INC_DIR: ${SFINDER_BROKERSERVER_INC_DIR}")

## GET SOURCES and HEADERS
set(THIS_SRCS ${THIS_SRCS} 
  ${SFINDER_BROKERSERVER_SRC_DIR}/SFinderBrokerClass.cpp 
  ${SFINDER_BROKERSERVER_SRC_DIR}/SFinderBroker.cpp 
  ${SFINDER_BROKERSERVER_SRC_DIR}/SFinderBrokerStateMachine.cpp 
  ${SFINDER_BROKERSERVER_SRC_DIR}/ClassFactory.cpp	
	${SFINDER_BROKERSERVER_SRC_DIR}/WorkerStateCallBack.cpp
	${SFINDER_BROKERSERVER_SRC_DIR}/WorkerSourceDataCallBack.cpp
	${SFINDER_BROKERSERVER_SRC_DIR}/SFinderBrokerDynAttrUtils.cpp
)

file(GLOB_RECURSE THIS_HEADERS ${SFINDER_BROKERSERVER_INC_DIR}/*.h)

set(THIS_HEADERS_WPATH "")
foreach (_headerFile ${THIS_HEADERS})
	get_filename_component(_basefilename ${_headerFile} NAME)
  list (APPEND THIS_HEADERS_WPATH ${_basefilename})
endforeach()
list(REMOVE_DUPLICATES THIS_HEADERS_WPATH)


MESSAGE(STATUS "THIS_SRCS: ${THIS_SRCS}")
MESSAGE(STATUS "THIS_HEADERS: ${THIS_HEADERS}")
MESSAGE(STATUS "THIS_HEADERS_WPATH: ${THIS_HEADERS_WPATH}")

## Append srcs and heads to project srcs and heads
list (APPEND SFINDER_BROKERSERVER_SRCS ${THIS_SRCS})
list (APPEND SFINDER_BROKERSERVER_HEADERS ${THIS_HEADERS})
list (APPEND SFINDER_BROKERSERVER_HEADERS_WPATH ${THIS_HEADERS_WPATH})
list (APPEND Main_SFINDER_BROKERSERVER_SRCS ${SFINDER_BROKERSERVER_SRC_DIR}/main.cpp)

set (SFINDER_BROKERSERVER_SRCS ${SFINDER_BROKERSERVER_SRCS} PARENT_SCOPE)
set (SFINDER_BROKERSERVER_HEADERS ${SFINDER_BROKERSERVER_HEADERS} PARENT_SCOPE)
set (SFINDER_BROKERSERVER_HEADERS_WPATH ${SFINDER_BROKERSERVER_HEADERS_WPATH} PARENT_SCOPE)
set (Main_SFINDER_BROKERSERVER_SRCS ${Main_SFINDER_BROKERSERVER_SRCS} PARENT_SCOPE)

list (APPEND INCLUDE_DIRS ${CMAKE_CURRENT_SOURCE_DIR})
set (INCLUDE_DIRS ${INCLUDE_DIRS} PARENT_SCOPE)
