## INC & SOURCE DIRS
set(SFINDER_SERVER_INC_DIR ${CMAKE_CURRENT_SOURCE_DIR})
set(SFINDER_SERVER_SRC_DIR ${CMAKE_CURRENT_SOURCE_DIR})
message (STATUS "SFINDER_SERVER_INC_DIR: ${SFINDER_SERVER_INC_DIR}")

## GET SOURCES and HEADERS
#file(GLOB_RECURSE THIS_SRCS ${LMCInterface_SRC_DIR}/*.cpp)
set(THIS_SRCS ${THIS_SRCS} 
  ${SFINDER_SERVER_SRC_DIR}/SFinderClass.cpp 
  ${SFINDER_SERVER_SRC_DIR}/SFinder.cpp 
  ${SFINDER_SERVER_SRC_DIR}/SFinderStateMachine.cpp 
  ${SFINDER_SERVER_SRC_DIR}/ClassFactory.cpp
  ${SFINDER_SERVER_SRC_DIR}/SFinderThread.cpp
)

file(GLOB_RECURSE THIS_HEADERS ${SFINDER_SERVER_INC_DIR}/*.h)

set(THIS_HEADERS_WPATH "")
foreach (_headerFile ${THIS_HEADERS})
	get_filename_component(_basefilename ${_headerFile} NAME)
  list (APPEND THIS_HEADERS_WPATH ${_basefilename})
endforeach()
list(REMOVE_DUPLICATES THIS_HEADERS_WPATH)


MESSAGE(STATUS "THIS_SRCS: ${THIS_SRCS}")
MESSAGE(STATUS "THIS_HEADERS: ${THIS_HEADERS}")
MESSAGE(STATUS "THIS_HEADERS_WPATH: ${THIS_HEADERS_WPATH}")

## Append srcs and heads to project srcs and heads
list (APPEND SFINDER_SERVER_SRCS ${THIS_SRCS})
list (APPEND SFINDER_SERVER_HEADERS ${THIS_HEADERS})
list (APPEND SFINDER_SERVER_HEADERS_WPATH ${THIS_HEADERS_WPATH})
list (APPEND Main_SFINDER_SERVER_SRCS ${SFINDER_SERVER_SRC_DIR}/main.cpp)

set (SFINDER_SERVER_SRCS ${SFINDER_SERVER_SRCS} PARENT_SCOPE)
set (SFINDER_SERVER_HEADERS ${SFINDER_SERVER_HEADERS} PARENT_SCOPE)
set (SFINDER_SERVER_HEADERS_WPATH ${SFINDER_SERVER_HEADERS_WPATH} PARENT_SCOPE)
set (Main_SFINDER_SERVER_SRCS ${Main_SFINDER_SERVER_SRCS} PARENT_SCOPE)

list (APPEND INCLUDE_DIRS ${CMAKE_CURRENT_SOURCE_DIR})
set (INCLUDE_DIRS ${INCLUDE_DIRS} PARENT_SCOPE)
